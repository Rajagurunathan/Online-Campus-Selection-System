CREATE PROCEDURE [dbo].[SELECT_SEARCH_COMPANY]

AS
BEGIN

end
else if @JOBCATE!='SELECT' and @QUALI!='SELECT' and @skill='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE JobCategory = @JOBCATE and Min_Qualification =@QUALI 
end
else if @JOBCATE!='SELECT' and @QUALI='SELECT' and @skill!='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE JobCategory = @JOBCATE   and RequiredSkill=@skill
end
else if @JOBCATE!='SELECT' and @QUALI='SELECT' and @skill='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE JobCategory = @JOBCATE
end 

else if @JOBCATE='SELECT' and @QUALI!='SELECT' and @skill!='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE RequiredSkill=@skill and Min_Qualification =@QUALI
end 
else if @JOBCATE='SELECT' and @QUALI!='SELECT' and @skill='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE Min_Qualification =@QUALI
end 
else if @JOBCATE='SELECT' and @QUALI='SELECT' and @skill!='SELECT'
begin
   SELECT * FROM  ADDJOBMST WHERE RequiredSkill=@skill
end 



END
